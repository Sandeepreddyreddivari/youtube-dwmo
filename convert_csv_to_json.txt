import csv
import json

def csv_to_json(file_one_csv_path, file_two_json_path):
    try:
        with open(file_one_csv_path, 'r') as f:
            data = csv.DictReader(f)
            data_up = list(data)
        with open(file_two_json_path, 'w') as f1:
            json.dump(data_up, f1, indent=4)
            print("Successfully converted CSV to JSON")
    except Exception as e:
        print(f'Could not convert the file because {e}')
